library(dplyr)
sales_data <- read.csv("C:/Users/farha/OneDrive/Desktop/R practs/archive/Walmart_Sales.csv")
print("First few rows of the sales data:")
head(sales_data)
print("Summary statistics for numeric columns:")
summary(sales_data)
total_sales_by_product <- sales_data %>%
group_by(Product.Name) %>%
summarise(Total_Sales_Amount = sum(Sales.Amount), Total_Quantity_Sold = sum(Quantity))
print("Total sales by product:")
print(total_sales_by_product)
mean_sales <- mean(sales_data$Sales.Amount)
median_sales <- median(sales_data$Sales.Amount)
sd_sales <- sd(sales_data$Sales.Amount)
sales_stats <- c(Mean_Sales = mean_sales, Median_Sales = median_sales, SD_Sales = sd_sales)
print("Descriptive statistics for Sales Amount:")
print(sales_stats)
total_sales_amount <- sum(sales_data$Sales.Amount)
total_quantity_sold <- sum(sales_data$Quantity)
print(paste("Total Sales Amount across all products:", total_sales_amount))
print(paste("Total Quantity Sold across all products:", total_quantity_sold))
max_sales_product <- sales_data[which.max(sales_data$Sales.Amount), ]
min_sales_product <- sales_data[which.min(sales_data$Sales.Amount), ]
print("Product with the highest sales:")
print(max_sales_product)
print("Product with the lowest sales:")
print(min_sales_product)
